```bash

# 循环遍历目录下文件，删除制定字符串。
for file in $(find . -mindepth 1 -type f); do sed -i 's/更多课程请加QQ群1707012297//g' $file; done

# 删除指定的字符，比如 'a'
sed -i 's/a//' file

# 移除每一行的第一个字符
sed -i 's/^.//' file

# 移除每一行的最后一个字符
sed -i 's/.$//' file

# 同时移除每一行的第一个字符和最后一个字符
# sed命令可以结合多个操作指令，每个指令之间通过';'分割
sed -i 's/^.//;s/.$//' file

# 移除每一行的匹配到的第一个字符，如果没有匹配不做处理
sed -i 's/^F//' file
# 以上命令移除了第一个且匹配F的字符

# 移除每一行的匹配到的最后一个字符，如果没有匹配不做处理
sed -i 's/x$//' file

# 删除每一行的第一到第三个字符
sed -i 's/...//' file

# 删除每一行的第一个到第 n 个字符
sed -i -r 's/.{n}//' file

# 删除每一行的最后 n 个字符
sed -i -r 's/.{3}$//' file

# 除了每一行的第一到第n个字符外，剩下的都要删除
sed -i -r 's/(.{3}).*/\1/' file
# .{3}匹配每一行开头3个字符，且用 () 分组；.* 表示匹配任意多个字符。在替换位通过 \1 表示保留第一个分组的内容。

# 删除每一行所有字符且保留结尾的n个字符
sed -i -r 's/.*(.{3})/\1/' file

# 删除每一行匹配到的多个字符
sed -i 's/[aoe]//g' file

# 删除每一行匹配到的第 n 个字符
sed -i 's/u//2' file

# 删除每一行以某个字符开头的子字符串
sed -i 's/a.*//' file

# 删除文件中每一行中出现的所有数字
sed -i 's/[0-9]//g' file

# 删除每一行中出现的小写字符
sed -i 's/[a-z]//g' file

# 删除除了小写字符之外的其他所有字符
sed -i 's/[^a-z]//g' file

# 删除每一行所有的字母和数字字符
sed -i 's/[a-z][A-Z][0-9]//g' file

# 删除每一行的一个字符，且忽略大小写
sed -i 's/[uU]//g'
```
